91 template <typename T>
92 dtParametrizedBehavior<T>::dtParametrizedBehavior(unsigned preAllocationSize)
93  : m_size(preAllocationSize)
94 {
95  if (m_size == 0)
96  m_agentsParams = 0;
97  else
98  {
99  m_agentsParams = (Node*) dtAlloc(sizeof(Node) * m_size, DT_ALLOC_PERM);
100 
101  for (unsigned i = 0; i < m_size; ++i)
102  {
103  new(&m_agentsParams[i]) Node();
104  m_agentsParams[i].next = 0;
105  m_agentsParams[i].id = UINT_MAX;
106  }
107  }
108 }

110 template <typename T>
111 dtParametrizedBehavior<T>::~dtParametrizedBehavior()
112 {
113  for (unsigned i = 0; i < m_size; ++i)
114  {
115  Node* n = &m_agentsParams[i];
116 
117  while (n->next)
118  {
119  Node* toDelete = n->next;
120  n->next = n->next->next;
121  dtFree(toDelete);
122  }
123  }
124 
125  dtFree(m_agentsParams);
126  m_agentsParams = 0;
127 }

129 template <typename T>
130 T* dtParametrizedBehavior<T>::getBehaviorParams(unsigned id) const
131 {
132  if (m_size == 0)
133  return 0;
134 
135  unsigned index = id % m_size;
136 
137  if (index >= m_size)
138  return 0;
139 
140  Node* head = &m_agentsParams[index];
141 
142  if (head->id == id && head->id < UINT_MAX)
143  return &head->value;
144 
145  while (head->next)
146  {
147  head = head->next;
148 
149  if (head->id == id && head->id < UINT_MAX)
150  return &head->value;
151  }
152 
153  void* mem = dtAlloc(sizeof(Node), DT_ALLOC_PERM);
154 
155  if (mem == 0)
156  return 0;
157 
158  Node* newNode = new(mem) Node;
159 
160  newNode->next = 0;
161  newNode->id = id;
162  head->next = newNode;
163 
164  return &newNode->value;
165 }
